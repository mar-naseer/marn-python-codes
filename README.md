## 💻 Example Program

# Program to find the LCM of two numbers

a = 6
b = 4
max_val = max(a, b)

for i in range(max_val, (a * b) + 1):
    if i % a == 0 and i % b == 0:
        print(f"The LCM of {a} and {b} is {i}")
        break
